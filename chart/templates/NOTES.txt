Thank you for installing {{ .Chart.Name }}!

Your release is named: {{ .Release.Name }}
Installed in namespace: {{ include "observability-dashboard.namespace" . }}

To check the deployment status:
  kubectl get pods -n {{ include "observability-dashboard.namespace" . }}

To access the service:
{{- if eq .Values.service.type "ClusterIP" }}
  kubectl port-forward -n {{ include "observability-dashboard.namespace" . }} service/{{ include "observability-dashboard.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get -n {{ include "observability-dashboard.namespace" . }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "observability-dashboard.fullname" . }})
  export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Service URL: http://$NODE_IP:$NODE_PORT"
{{- else if eq .Values.service.type "LoadBalancer" }}
  export SERVICE_IP=$(kubectl get svc -n {{ include "observability-dashboard.namespace" . }} {{ include "observability-dashboard.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Service URL: http://$SERVICE_IP:{{ .Values.service.port }}"
{{- end }}

Health check endpoint:
  http://localhost:{{ .Values.service.port }}/health

To view logs:
  kubectl logs -n {{ include "observability-dashboard.namespace" . }} -l app=observability-dashboard -f
